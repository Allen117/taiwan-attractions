{%- comment -%} Renders the clickable Taiwan SVG map. Optional include params: active: county slug to highlight zoom: "true" to zoom viewBox to the active county's bbox {%- endcomment -%} {%- assign active = include.active -%} {%- assign viewbox = "0 0 1440 2055" -%} {%- if active and include.zoom -%} {%- assign ac = site.data.counties | where: "slug", active | first -%} {%- if ac.bbox -%} {%- assign pad = 40 -%} {%- assign vx = ac.bbox[0] | minus: pad -%} {%- assign vy = ac.bbox[1] | minus: pad -%} {%- assign vw = ac.bbox[2] | plus: pad | plus: pad -%} {%- assign vh = ac.bbox[3] | plus: pad | plus: pad -%} {%- assign viewbox = vx | append: " " | append: vy | append: " " | append: vw | append: " " | append: vh -%} {%- endif -%} {%- endif -%}
